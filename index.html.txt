<div id="reader"></div>

<script>
  const tg = window.Telegram.WebApp;
  tg.expand();

  const qrCodeScanner = new Html5Qrcode("reader");

  qrCodeScanner.start(
    { facingMode: "environment" },
    { fps: 10, qrbox: 250 },
    (decodedText) => {
      tg.sendData(decodedText); // send sheet link to bot
      qrCodeScanner.stop();
    }
  );
</script>
